# Pantavisor Initial Images (Yocto)

We provide images generated by our CI using the Yocto project.

Here, you can download our flashable base Pantavisor image for your device. For instructions on how to install one of these Pantavisor images, see our [how-to guide](choose-image.md).

Note that the images in these lists are generic and will later require the device to be [claimed](claim-device.md) if you want to interact with it from [Pantacor Hub](remote-control.md#pantacor-hub).

The Initial Images are referred to as `pantavisor-starter`, which includes the following components:

*   **BSP**: Includes Pantavisor, kernel, device-trees, and u-boot files.
*   **Containers**:
    *   `pv-alpine-connman`: Networking management.
    *   `pv-pvr-sdk`: Container with tooling to manage Pantavisor.
    *   `pv-wificonnect`: WiFi captive portal to help configuring WiFi-enabled boards.

The initial images are stored at [releases.json](https://pantavisor-ci.s3.amazonaws.com/meta-pantavisor/releases.json).

<script>
(function() {
    const JSON_URL = 'https://pantavisor-ci.s3.amazonaws.com/meta-pantavisor/releases.json';

    function renderTable(containerId, versionsObj) {
        const container = document.getElementById(containerId);
        if (!versionsObj || Object.keys(versionsObj).length === 0) {
            container.innerHTML = '<p>No images available at the moment.</p>';
            return;
        }

        // Filter out timestamp and get the latest version by sorting keys
        const versions = Object.keys(versionsObj)
            .filter(v => v !== 'timestamp')
            .sort((a, b) => {
                return b.localeCompare(a, undefined, { numeric: true, sensitivity: 'base' });
            });

        if (versions.length === 0) {
            container.innerHTML = '<p>No versioned images found.</p>';
            return;
        }

        const latestVersion = versions[0];
        const images = versionsObj[latestVersion];

        let html = `<h3>Version: ${latestVersion}</h3>`;
        html += '<div class="md-typeset__table"><table><thead><tr>';
        html += '<th>Device</th><th>Full Image</th><th>BSP(pantavisor)</th><th>SDK</th><th>SHA256 (Full Image)</th>';
        html += '</tr></thead><tbody>';

        let hasImages = false;
        images.forEach(img => {
            if (!img.name) return;
            hasImages = true;

            const fullImageLink = img.full_image ? `<a href="${img.full_image.url}">Download</a>` : 'N/A';
            const bspLink = img.bsp ? `<a href="${img.bsp.url}">Download</a>` : 'N/A';
            const sdkLink = img.sdk ? `<a href="${img.sdk.url}">Download</a>` : 'N/A';
            const sha256 = img.full_image && img.full_image.sha256 ? `<code>${img.full_image.sha256}</code>` : 'N/A';

            html += `<tr>
                <td><strong>${img.name}</strong></td>
                <td>${fullImageLink}</td>
                <td>${bspLink}</td>
                <td>${sdkLink}</td>
                <td>${sha256}</td>
            </tr>`;
        });

        if (!hasImages) {
            container.innerHTML = '<p>No images found for the latest version.</p>';
            return;
        }

        html += '</tbody></table></div>';
        container.innerHTML = html;
    }

    fetch(JSON_URL)
        .then(response => response.json())
        .then(data => {
            renderTable('stable-images-container', data.stable);
            renderTable('rc-images-container', data['release-candidate']);
        })
        .catch(error => {
            console.error('Error fetching releases:', error);
            document.getElementById('stable-images-container').innerHTML = '<p>Error loading images. Please check your internet connection or try again later.</p>';
            document.getElementById('rc-images-container').innerHTML = '';
        });
})();
</script>
